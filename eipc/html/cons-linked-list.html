function cons(data, lst) {
  let head = new Node(data);
  head.next = lst;
  return head;
}

function Node(data) {
  this.data = data;
  
  this.disp = () => {
    let start = this;
    while (start) {
      console.log(start.data);
      start = start.next;
    }
  }
  
  this.dispES5 = function() {
    var start = this;
    while (start) {
      console.log(start.data);
      start = start.next;
    }
  }
}

function car(lst) {
  return lst.data;
}

function cdr(lst) {
  return lst.next;
}

let lst = null;
lst = cons(3, lst);
lst = cons(4, lst);
lst = cons(5, lst);
lst.dispES5();

console.log(car(lst));
cdr(lst).disp();
