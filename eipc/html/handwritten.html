class Grandparent {
  constructor(name) {
    this.name = name;
  }
  doSomething(x) {
    return "grandparent does " + x;
  }
}

class Parent extends Grandparent {
  constructor(name) {
    super(name);
  }
}

class MyClass extends Parent {
  constructor(name) {
    super(name);
  }
  doSomething(x) {
    return super.doSomething(x) + " with grandparent";
  }
}
