function drawDragon(length) {
  if (length > 4) {
    rt(270);

    drawDragon(length/2);
    rt(135);

    drawDragon(Math.sqrt(2 * (length/2) * (length/2)));
    rt(360-45);

    jump(length/2);

    rt(180);
    drawDragon(length/2);

    rt(180);
    jump(length/2);
  } else {
    fd(length);
  }
}
